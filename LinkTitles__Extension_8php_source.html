129  private static function parseContent( $title, &$text ) {
130 
131  // Configuration variables need to be defined here as globals.
132  global $wgLinkTitlesFirstOnly;
133  global $wgLinkTitlesSmartMode;
134  global $wgCapitalLinks;
135 
136  ( $wgLinkTitlesFirstOnly ) ? $limit = 1 : $limit = -1;
137  $limitReached = false;
138  self::$currentTitle = $title;
139  $currentNamespace = $title->getNamespace();
140  $newText = $text;
141 
142  if ( !isset( self::$pageTitles ) || ( $currentNamespace != self::$currentNamespace ) ) {
143  self::$currentNamespace = $currentNamespace;
144  self::fetchPageTitles();
145  }
146 
147  // Iterate through the page titles
148  foreach( self::$pageTitles as $row ) {
149  self::newTarget( $row->page_namespace, $row->page_title );
150 
151  // Don't link current page
152  if ( self::$targetTitle->equals( self::$currentTitle ) ) { continue; }
153 
154  // split the page content by [[...]] groups
155  // credits to inhan @ StackOverflow for suggesting preg_split
156  // see http://stackoverflow.com/questions/10672286
157  $arr = preg_split( self::$delimiter, $newText, -1, PREG_SPLIT_DELIM_CAPTURE );
158 
159  // Escape certain special characters in the page title to prevent
160  // regexp compilation errors
161  self::$targetTitleText = self::$targetTitle->getText();
162  $quotedTitle = preg_quote( self::$targetTitleText, '/' );
163 
164  self::ltDebugLog( 'TargetTitle='. self::$targetTitleText, 'private' );
165  self::ltDebugLog( 'TargetTitleQuoted='. $quotedTitle, 'private' );
166 
167  // Depending on the global configuration setting $wgCapitalLinks,
168  // the title has to be searched for either in a strictly case-sensitive
169  // way, or in a 'fuzzy' way where the first letter of the title may
170  // be either case.
171  if ( $wgCapitalLinks && ( $quotedTitle[0] != '\\' )) {
172  $searchTerm = '((?i)' . $quotedTitle[0] . '(?-i)' .
173  substr($quotedTitle, 1) . ')';
174  } else {
175  $searchTerm = '(' . $quotedTitle . ')';
176  }
177 
178  $regex = '/(?<![\:\.\@\/\?\&])' . self::$wordStartDelim .
179  $searchTerm . self::$wordEndDelim . '/S';
180  for ( $i = 0; $i < count( $arr ); $i+=2 ) {
181  // even indexes will point to text that is not enclosed by brackets
182  $arr[$i] = preg_replace_callback( $regex,
183  'LinkTitles\Extension::simpleModeCallback', $arr[$i], $limit, $count );
184  if ( $wgLinkTitlesFirstOnly && ( $count > 0 ) ) {
185  $limitReached = true;
186  break;
187  };
188  };
189  $newText = implode( '', $arr );
190 
191  // If smart mode is turned on, the extension will perform a second
192  // pass on the page and add links with aliases where the case does
193  // not match.
194  if ( $wgLinkTitlesSmartMode && !$limitReached ) {
195  $arr = preg_split( self::$delimiter, $newText, -1, PREG_SPLIT_DELIM_CAPTURE );
196 
197  for ( $i = 0; $i < count( $arr ); $i+=2 ) {
198  // even indexes will point to text that is not enclosed by brackets
199  $arr[$i] = preg_replace_callback( '/(?<![\:\.\@\/\?\&])' .
200  self::$wordStartDelim . '(' . $quotedTitle . ')' .
201  self::$wordEndDelim . '/iS', 'LinkTitles\Extension::smartModeCallback',
202  $arr[$i], $limit, $count );
203  if ( $wgLinkTitlesFirstOnly && ( $count > 0 )) {
204  break;
205  };
206  };
207  $newText = implode( '', $arr );
208  } // $wgLinkTitlesSmartMode
209  }; // foreach $res as $row
210  return $newText;
211  }

278  private static function fetchPageTitles() {
279  global $wgLinkTitlesPreferShortTitles;
280  global $wgLinkTitlesMinimumTitleLength;
281  global $wgLinkTitlesBlackList;
282  global $wgLinkTitlesNamespaces;
283 
284  ( $wgLinkTitlesPreferShortTitles ) ? $sort_order = 'ASC' : $sort_order = 'DESC';
285  // Build a blacklist of pages that are not supposed to be link
286  // targets. This includes the current page.
287  $blackList = str_replace( ' ', '_', '("' . implode( '","',$wgLinkTitlesBlackList ) . '")' );
288 
289  // Build our weight list. Make sure current namespace is first element
290  $namespaces = array_diff( $wgLinkTitlesNamespaces, [ self::$currentNamespace ] );
291  array_unshift( $namespaces, self::$currentNamespace );
292 
293  // No need for sanitiy check. we are sure that we have at least one element in the array
294  $weightSelect = "CASE page_namespace ";
295  $currentWeight = 0;
296  foreach ($namespaces as &$namspacevalue) {
297  $currentWeight = $currentWeight + 100;
298  $weightSelect = $weightSelect . " WHEN " . $namspacevalue . " THEN " . $currentWeight . PHP_EOL;
299  }
300  $weightSelect = $weightSelect . " END ";
301  $namespacesClause = '(' . implode( ', ', $namespaces ) . ')';
302 
303  // Build an SQL query and fetch all page titles ordered by length from
304  // shortest to longest. Only titles from 'normal' pages (namespace uid
305  // = 0) are returned. Since the db may be sqlite, we need a try..catch
306  // structure because sqlite does not support the CHAR_LENGTH function.
307  $dbr = wfGetDB( DB_SLAVE );
308  try {
309  $res = $dbr->select(
310  'page',
311  array( 'page_title', 'page_namespace' , "weight" => $weightSelect),
312  array(
313  'page_namespace IN ' . $namespacesClause,
314  'CHAR_LENGTH(page_title) >= ' . $wgLinkTitlesMinimumTitleLength,
315  'page_title NOT IN ' . $blackList,
316  ),
317  __METHOD__,
318  array( 'ORDER BY' => 'weight ASC, CHAR_LENGTH(page_title) ' . $sort_order )
319  );
320  } catch (Exception $e) {
321  $res = $dbr->select(
322  'page',
323  array( 'page_title', 'page_namespace' , "weight" => $weightSelect ),
324  array(
325  'page_namespace IN ' . $namespacesClause,
326  'LENGTH(page_title) >= ' . $wgLinkTitlesMinimumTitleLength,
327  'page_title NOT IN ' . $blackList,
328  ),
329  __METHOD__,
330  array( 'ORDER BY' => 'weight ASC, LENGTH(page_title) ' . $sort_order )
331  );
332  }
333 
334  self::$pageTitles = $res;
335  return true;
336  }

350  // Callback function for use with preg_replace_callback.
351  // This essentially performs a case-sensitive comparison of the
352  // current page title and the occurrence found on the page; if
353  // the cases do not match, it builds an aliased (piped) link.
354  // If $wgCapitalLinks is set to true, the case of the first
355  // letter is ignored by MediaWiki and we don't need to build a
356  // piped link if only the case of the first letter is different.
357  private static function smartModeCallback( array $matches ) {
358  global $wgCapitalLinks;
359 
360  if ( $wgCapitalLinks ) {
361  // With $wgCapitalLinks set to true we have a slightly more
362  // complicated version of the callback than if it were false;
363  // we need to ignore the first letter of the page titles, as
364  // it does not matter for linking.
365  if ( self::checkTargetPage() ) {
366  self::ltLog( "Linking (smart) '$matches[0]' to '" . self::$targetTitle . "'" );
367  if ( strcmp(substr(self::$targetTitleText, 1), substr($matches[0], 1)) == 0 ) {
368  // Case-sensitive match: no need to bulid piped link.
369  return '[[' . $matches[0] . ']]';
370  } else {
371  // Case-insensitive match: build piped link.
372  return '[[' . self::$targetTitleText . '|' . $matches[0] . ']]';
373  }
374  }
375  else
376  {
377  return $matches[0];
378  }
379  } else {
380  // If $wgCapitalLinks is false, we can use the simple variant
381  // of the callback function.
382  if ( self::checkTargetPage() ) {
383  self::ltLog( "Linking (smart) '$matches[0]' to '" . self::$targetTitle . "'" );
384  if ( strcmp(self::$targetTitleText, $matches[0]) == 0 ) {
385  // Case-sensitive match: no need to bulid piped link.
386  return '[[' . $matches[0] . ']]';
387  } else {
388  // Case-insensitive match: build piped link.
389  return '[[' . self::$targetTitleText . '|' . $matches[0] . ']]';
390  }
391  }
392  else
393  {
394  return $matches[0];
395  }
396  }
397  }

428  private static function checkTargetPage() {
429  global $wgLinkTitlesEnableNoTargetMagicWord;
430  global $wgLinkTitlesCheckRedirect;
431 
432  // If checking for redirects is enabled and the target page does
433  // indeed redirect to the current page, return the page title as-is
434  // (unlinked).
435  if ( $wgLinkTitlesCheckRedirect ) {
436  $redirectTitle = self::getTargetContent()->getUltimateRedirectTarget();
437  if ( $redirectTitle && $redirectTitle->equals(self::$currentTitle) ) {
438  return false;
439  }
440  };
441 
442  // If the magic word __NOAUTOLINKTARGET__ is enabled and the target
443  // page does indeed contain this magic word, return the page title
444  // as-is (unlinked).
445  if ( $wgLinkTitlesEnableNoTargetMagicWord ) {
446  if ( self::getTargetContent()->matchMagicWord(
447  \MagicWord::get('MAG_LINKTITLES_NOTARGET') ) ) {
448  return false;
449  }
450  };
451  return true;
452  }

457  private static function BuildDelimiters() {
458  // Configuration variables need to be defined here as globals.
459  global $wgLinkTitlesParseHeadings;
460  global $wgLinkTitlesSkipTemplates;
461  global $wgLinkTitlesWordStartOnly;
462  global $wgLinkTitlesWordEndOnly;
463 
464  // Use unicode character properties rather than \b escape sequences
465  // to detect whole words containing non-ASCII characters as well.
466  // Note that this requires a PCRE library that was compiled with
467  // --enable-unicode-properties
468  ( $wgLinkTitlesWordStartOnly ) ? self::$wordStartDelim = '(?<!\pL)' : self::$wordStartDelim = '';
469  ( $wgLinkTitlesWordEndOnly ) ? self::$wordEndDelim = '(?!\pL)' : self::$wordEndDelim = '';
470 
471  if ( $wgLinkTitlesSkipTemplates )
472  {
473  // Use recursive regex to balance curly braces;
474  // see http://www.regular-expressions.info/recurse.html
475  $templatesDelimiter = '{{(?>[^{}]|(?R))*}}|';
476  } else {
477  // Match template names (ignoring any piped [[]] links in them)
478  // along with the trailing pipe and parameter name or closing
479  // braces; also match sequences of '|wordcharacters=' (without
480  // spaces in them) that usually only occur as parameter names in
481  // transclusions (but could also occur as wiki table cell contents).
482  // TODO: Find a way to match parameter names in transclusions, but
483  // not in table cells or other sequences involving a pipe character
484  // and equal sign.
485  $templatesDelimiter = '{{[^|]*?(?:(?:\[\[[^]]+]])?)[^|]*?(?:\|(?:\w+=)?|(?:}}))|\|\w+=|';
486  }
487 
488  // Build a regular expression that will capture existing wiki links ("[[...]]"),
489  // wiki headings ("= ... =", "== ... ==" etc.),
490  // urls ("http://example.com", "[http://example.com]", "[http://example.com Description]",
491  // and email addresses ("mail@example.com").
492  // Since there is a user option to skip headings, we make this part of the expression
493  // optional. Note that in order to use preg_split(), it is important to have only one
494  // capturing subpattern (which precludes the use of conditional subpatterns).
495  ( $wgLinkTitlesParseHeadings ) ? $delimiter = '' : $delimiter = '=+.+?=+|';
496  $urlPattern = '[a-z]+?\:\/\/(?:\S+\.)+\S+(?:\/.*)?';
497  self::$delimiter = '/(' . // exclude from linking:
498  '\[\[.*?\]\]|' . // links
499  $delimiter . // titles (if requested)
500  $templatesDelimiter . // templates (if requested)
501  '^ .+?\n|\n .+?\n|\n .+?$|^ .+?$|' . // preformatted text
502  '<nowiki>.*?<.nowiki>|<code>.*?<\/code>|' . // nowiki/code
503  '<pre>.*?<\/pre>|<html>.*?<\/html>|' . // pre/html
504  '<script>.*?<\/script>|' . // script
505  '<gallery>.*?<\/gallery>|' . // gallery
506  '<div.+?>|<\/div>|' . // attributes of div elements
507  '<span.+?>|<\/span>|' . // attributes of span elements
508  '<file>[^<]*<\/file>|' . // stuff inside file elements
509  'style=".+?"|class=".+?"|' . // styles and classes (e.g. of wikitables)
510  '<noautolinks>.*?<\/noautolinks>|' . // custom tag 'noautolinks'
511  '\[' . $urlPattern . '\s.+?\]|'. $urlPattern . '(?=\s|$)|' . // urls
512  '(?<=\b)\S+\@(?:\S+\.)+\S+(?=\b)' . // email addresses
513  ')/ismS';
514  }
